// Feature kontrolü
const result = await isFeatureEnabled(businessId, FEATURES.REPORTS_ADVANCED);
if (result.enabled) {
  // Show advanced reports
}

// Feature assert (throw if disabled)
await assertFeatureEnabled(businessId, FEATURES.INTEGRATIONS_API);
// API key oluşturma devam eder

// Feature limit al
const tableLimit = await getFeatureLimit(businessId, FEATURES.TABLES_LIMIT);
console.log(`Table limit: ${tableLimit}`);

// Multiple feature check
const features = await checkFeatures(businessId, [
  FEATURES.REPORTS_ADVANCED,
  FEATURES.BRANDING_CUSTOM,
  FEATURES.INTEGRATIONS_API,
]);

📊 USAGE TRACKING KULLANIMI

// Usage limit kontrolü
const check = await checkUsageLimit(businessId, 'tables');
if (check.exceeded) {
  throw new Error(`Table limit exceeded: ${check.current}/${check.limit}`);
}

// Usage limit assert (throw if exceeded)
await assertUsageLimit(businessId, 'products');
// Ürün oluşturma devam eder

// Usage warning kontrolü (soft limit)
const warning = await checkUsageWarning(businessId, 'orders', 80);
if (warning.shouldWarn) {
  console.log(warning.message);
  // Show upgrade prompt
}

// Usage metric increment
await incrementUsageMetric(businessId, 'ORDERS_COUNT');
await incrementUsageMetric(businessId, 'API_CALLS_COUNT');

// Usage summary
const summary = await getUsageSummary(businessId);
console.log(`Tables: ${summary.tables.current}/${summary.tables.limit}`);
console.log(`Orders: ${summary.orders.current}/${summary.orders.limit}`);

  GÜVENLİK CONSIDERATIONS
═══════════════════════════════════════════════════════════════════════════════

🔐 API KEY SECURITY

1. Key Generation:
   • Cryptographically secure random
   • Format: qrm_live_xxxxxxxxxxxxx
   • Store only hash (bcrypt)

2. Key Rotation:
   • Support multiple active keys
   • Graceful deprecation period
   • Automatic expiration

3. Rate Limiting:
   • Per-key rate limits
   • Scope-based limits
   • Burst protection

🔐 WEBHOOK SECURITY

1. Signature Verification:
   • HMAC-SHA256 signature
   • Timestamp validation
   • Replay attack prevention

2. Retry Strategy:
   • Exponential backoff
   • Max 5 attempts
   • Dead letter queue

3. Payload Validation:
   • Schema validation
   • Size limits (1MB)
   • Content-type check

🔐 BILLING SECURITY

1. Provider Abstraction:
   • No hardcoded credentials
   • Environment variables
   • Encrypted config storage

2. Webhook Verification:
   • Provider signature check
   • Idempotency keys
   • Event deduplication

3. Invoice Security:
   • Tenant-scoped access
   • Audit trail
   • PCI compliance (if storing cards)

📝 DATA MIGRATION

-- Add default values for existing businesses
UPDATE businesses 
SET 
  currency = 'TRY',
  locale = 'tr',
  timezone = 'Europe/Istanbul',
  orderMode = 'TABLE_SERVICE',
  waiterCallEnabled = true,
  paymentRequestEnabled = true
WHERE currency IS NULL;

-- Set limits based on current usage
UPDATE businesses b
SET 
  maxTables = (SELECT COUNT(*) FROM tables WHERE businessId = b.id) + 5,
  maxWaiters = (SELECT COUNT(*) FROM users WHERE businessId = b.id AND role = 'WAITER') + 2,
  maxProducts = (SELECT COUNT(*) FROM products WHERE businessId = b.id) + 20,
  maxMonthlyOrders = 500
WHERE maxTables IS NULL;
